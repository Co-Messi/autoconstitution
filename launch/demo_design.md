### Live Demo Code

```python
# demo_single_agent.py
"""Single-agent baseline for comparison"""
import time
from single_agent import SingleAgentResearch

start = time.time()
agent = SingleAgentResearch(model="gpt-4")
result = agent.research(
    "Analyze the competitive landscape of AI agent frameworks in 2025"
)
elapsed = time.time() - start

print(f"Time: {elapsed:.1f}s")
print(f"Output length: {len(result)} chars")
print(f"Sources: {count_sources(result)}")
```

```python
# demo_swarm.py
"""SwarmResearch demonstration"""
import time
import asyncio
from swarmresearch import SwarmOrchestrator

async def main():
    start = time.time()
    
    orchestrator = SwarmOrchestrator(
        director_model="claude-3-5-sonnet",
        worker_model="gpt-4",
        max_workers=6,
        enable_cross_pollination=True,
        enable_critic=True
    )
    
    result = await orchestrator.research(
        "Analyze the competitive landscape of AI agent frameworks in 2025"
    )
    
    elapsed = time.time() - start
    
    print(f"⏱️  Time: {elapsed:.1f}s")
    print(f"📊 Coverage: {result.coverage_score}%")
    print(f"🔍 Sources: {len(result.sources)}")
    print(f"🐝 Agents used: {result.agent_count}")
    print(f"🔄 Cross-pollination events: {result.pollination_count}")
    
    return result

asyncio.run(main())
```

### Visual Dashboard Components

```python
# demo_dashboard.py
"""Real-time dashboard for demo visualization"""
from rich.live import Live
from rich.table import Table
from rich.panel import Panel
from rich.layout import Layout

class DemoDashboard:
    def __init__(self):
        self.layout = Layout()
        self.layout.split_column(
            Layout(name="header", size=3),
            Layout(name="main"),
            Layout(name="footer", size=5)
        )
        self.layout["main"].split_row(
            Layout(name="left"),
            Layout(name="right")
        )
    
    def update_agents(self, agents):
        """Update agent status display"""
        table = Table(title="Active Agents")
        table.add_column("Agent", style="cyan")
        table.add_column("Task", style="green")
        table.add_column("Status", style="yellow")
        table.add_column("Progress", style="magenta")
        
        for agent in agents:
            table.add_row(
                agent.name,
                agent.task[:30] + "...",
                agent.status,
                f"{agent.progress}%"
            )
        
        self.layout["left"].update(Panel(table))
    
    def update_bus(self, events):
        """Update cross-pollination bus display"""
        content = "\n".join([
            f"[{e.timestamp}] {e.source} → ALL: {e.message[:50]}"
            for e in events[-5:]
        ])
        self.layout["right"].update(Panel(
            content, 
            title="Cross-Pollination Bus",
            border_style="blue"
        ))
    
    def update_metrics(self, metrics):
        """Update metrics display"""
        content = f"""
⏱️  Time: {metrics.elapsed:.1f}s
📊 Coverage: {metrics.coverage}%
🔍 Sources: {metrics.sources}
🐝 Active Agents: {metrics.active_agents}
🔄 Pollination Events: {metrics.pollinations}
        """
        self.layout["footer"].update(Panel(
            content,
            title="Live Metrics",
            border_style="green"
        ))
```
